import streamlit as st
import pandas as pd
import plotly.express as px


def normalize_columns(df: pd.DataFrame) -> pd.DataFrame:
    df = df.copy()
    df.columns = (
        df.columns.astype(str)
        .str.strip()
        .str.lower()
        .str.replace(" ", "_", regex=False)
    )
    return df


def find_column(df_cols, aliases):
    cols_set = set(df_cols)
    for a in aliases:
        if a in cols_set:
            return a
    for a in aliases:
        tokens = [t for t in a.split("_") if t]
        for c in df_cols:
            if all(t in c for t in tokens):
                return c
    return None


def main():
    st.set_page_config(layout="wide")
    st.title("Provisional Natality Data Dashboard")
    st.subheader("Birth Analysis by State and Gender")

    try:
        df = pd.read_csv("Provisional_Natality_2025_CDC.csv")
    except FileNotFoundError:
        st.error("Dataset file not found in repository.")
        return
    except Exception as e:
        st.error("An error occurred while loading the dataset.")
        st.write(str(e))
        return

    df = normalize_columns(df)

    required_logical_fields = [
        "state_of_residence",
        "month",
        "month_code",
        "year_code",
        "sex_of_infant",
        "births",
    ]

    alias_map = {
        "state_of_residence": [
            "state_of_residence",
            "state_residence",
            "residence_state",
            "state",
            "state_name",
            "state_of_res",
            "stateofresidence",
            "state_of_residence_name",
        ],
        "month": [
            "month",
            "month_name",
            "month_of_birth",
            "birth_month",
        ],
        "month_code": [
            "month_code",
            "month_cd",
            "month_num",
            "month_number",
            "monthid",
        ],
        "year_code": [
            "year_code",
            "year_cd",
            "year",
            "year_num",
            "year_number",
            "yearid",
        ],
        "sex_of_infant": [
            "sex_of_infant",
            "infant_sex",
            "sex",
            "sex_of_child",
            "sex_of_baby",
            "gender",
            "infant_gender",
        ],
        "births": [
            "births",
            "birth_count",
            "births_count",
            "number_of_births",
            "total_births",
            "count",
        ],
    }

    mapped_cols = {}
    missing = []
    for logical in required_logical_fields:
        col = find_column(df.columns.tolist(), alias_map.get(logical, [logical]))
        if col is None:
            missing.append(logical)
        else:
            mapped_cols[logical] = col

    if missing:
        st.error(
            "Missing required logical fields: "
            + ", ".join(missing)
            + ". Please verify the dataset column names."
        )
        st.write(df.columns)
        return

    df_work = df.copy()

    df_work[mapped_cols["births"]] = pd.to_numeric(df_work[mapped_cols["births"]], errors="coerce")
    df_work = df_work.dropna(subset=[mapped_cols["births"]])

    st.sidebar.header("Filters")

    state_col = mapped_cols["state_of_residence"]
    month_col = mapped_cols["month"]
    sex_col = mapped_cols["sex_of_infant"]

    state_options = sorted([v for v in df_work[state_col].dropna().unique().tolist()])
    month_options = sorted([v for v in df_work[month_col].dropna().unique().tolist()])
    sex_options = sorted([v for v in df_work[sex_col].dropna().unique().tolist()])

    state_sel = st.sidebar.multiselect(
        "State of Residence",
        options=["All"] + state_options,
        default=["All"],
    )
    month_sel = st.sidebar.multiselect(
        "Month",
        options=["All"] + month_options,
        default=["All"],
    )
    sex_sel = st.sidebar.multiselect(
        "Gender",
        options=["All"] + sex_options,
        default=["All"],
    )

    filtered = df_work.copy()

    if state_sel and "All" not in state_sel:
        filtered = filtered[filtered[state_col].isin(state_sel)]
    if month_sel and "All" not in month_sel:
        filtered = filtered[filtered[month_col].isin(month_sel)]
    if sex_sel and "All" not in sex_sel:
        filtered = filtered[filtered[sex_col].isin(sex_sel)]

    if filtered.empty:
        st.warning("No data available for the selected filters.")
        return

    births_col = mapped_cols["births"]
    agg = (
        filtered.groupby([state_col, sex_col], dropna=False)[births_col]
        .sum()
        .reset_index()
        .rename(
            columns={
                state_col: "state_of_residence",
                sex_col: "sex_of_infant",
                births_col: "births",
            }
        )
    )

    agg["state_of_residence"] = agg["state_of_residence"].astype(str)
    agg = agg.sort_values(by=["state_of_residence", "sex_of_infant"], kind="mergesort")

    fig = px.bar(
        agg,
        x="state_of_residence",
        y="births",
        color="sex_of_infant",
        title="Total Births by State and Gender",
        labels={
            "state_of_residence": "State of Residence",
            "births": "Total Births",
            "sex_of_infant": "Gender",
        },
    )

    fig.update_layout(
        template="plotly_white",
        legend_title_text="Gender",
        autosize=True,
        margin=dict(l=10, r=10, t=60, b=10),
    )

    fig.update_xaxes(categoryorder="category ascending")
    st.plotly_chart(fig, use_container_width=True)

    df_display = filtered.reset_index(drop=True)

    try:
        st.dataframe(df_display, use_container_width=True, hide_index=True)
    except TypeError:
        st.dataframe(df_display, use_container_width=True)


if __name__ == "__main__":
    main()
